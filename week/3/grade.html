const score = +prompt("점수를 입력하세요");
    const grade =
      score < 0 || score > 100
        ? false
        : score >= 90
        ? "A"
        : score >= 80
        ? "B"
        : score >= 70
        ? "C"
        : score >= 60
        ? "D"
        : "F";
    document.write(
      !grade
        ? "올바르지 않은 점수를 입력하였습니다. 다시 입력해주세요."
        : `당신의 점수는 ${score}점이고, 학점은 ${grade}입니다.`
    );
