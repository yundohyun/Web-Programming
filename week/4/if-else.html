    let month = +prompt("월 입력");
    let season = null;
    // 이렇게 쓰지 마세요
    if (month > 2 && month < 9) {
      if (month < 6) {
        season = "봄";
      } else {
        season = "여름";
      }
    } else {
      if (month > 2 && month < 12) {
        season = "가을";
      } else {
        season = "겨울";
      }
    }
    document.write(`현재 날씨는 ${season}입니다.`);
