끝말잇기

      
    <div>
      <span id="num"></span>명이 끝말잇기를 시작합니다.
      <hr />
    </div>
    <style>
      body > div span {
        color: red;
        font-size: 24px;
        border: 2px blue dotted;
        padding-inline: 3px;
      }
    </style>
    <script>
      let answer;
      while (!answer) {
        const num = +prompt("끝말잇기를 할 인원을 입력하세요.");
        if (isNaN(num) || !num) continue;
        document.querySelector("#num").textContent = num;
        answer = confirm(`${num}명이 맞습니까?`);
      }
    </script>
명이 끝말잇기를 시작합니다.
