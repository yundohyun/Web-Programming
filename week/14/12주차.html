    <script>
      // 1. 다음 id로 p 태그를 script태그로 불러오도록 프로그래밍하시오.
      const p = document.getElementById("p");

      // 2. over함수는 클릭시 p태그의  배경색을 red로 표현하도록 프로그래밍하시오.
      function over() {
        p.style.backgroundColor = "red";
      }

      // 3. 2번의 over()함수를 에로우함수로 변경하시오.
      over = () => {
        p.style.backgroundColor = "red";
      };

      // 4. h2 태그을 자바스트립트로 불러와 header문자열 대신 javascript로 변경하시오.
      document.querySelector("h2").textContent = "javascript";

      // 5. 4번의 연장으로 h2태그에 대해 글자색을 red로 변경하시오.
      document.querySelector("h2").style.color = "red";

      // 6. 다음 change함수를 작성하시오.
      /*
        <button onclick="change(this,'skyblue','30px','왕자다')">나는요!!</button>
      */
      function change(btn, color, size, text) {
        btn.style.backgroundColor = color;
        btn.style.fontSize = size;
        btn.textContent = text;
      }

      // 7. 다음 out함수는 마우스가 밖으로 빠져나갈 경우
      // 현디렉토리에 있는 cat.png 이미지가 보이도록 한다.
      // 이에 맞게 프로그래밍하시오.
      /*
        <div id="div" onmouseout="out()">I love winter.</div>
      */
      function out() {
        document.getElementById("div").innerHTML =
          '<img src="cat.png" alt="cat" />';
      }

      // 8. 다음은 button 클릭시 질문이 답이 나오도록 answer 함수를 프로그래밍하시오.
      /*
        <button onclick="answer(this,'백설공주')">답보기</button>
      */
      function answer(btn, answer) {
        btn.textContent = answer;
      }

      // 9. 8번의 onclick함수를 addEventListener를 이용하여 프로그래밍하시오.
      document.querySelector("button").addEventListener("click", () => {
        answer(this, "백설공주");
      });
    </script>
