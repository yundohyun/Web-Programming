    // 다음 소스를 보고 물음에 답하시오
    const nums = [];
    for (let i = 0; i < 10; i++) nums.push(i);
  
    // 1. 다음 nums의 각 요소에 map 함수를 이용하여 5씩 더하도록 프로그래밍 하시오.
    nums.map((num) => num + 5);
  
    // 2. nums에서 홀수인 값만 돌려받는 배열 메소드를 이용하여 프로그래밍하시오. 단 한 개값만 돌려받도록 한다.
    nums.find((num) => num % 2 == 1);
  
    // 3. nums에서 홀수인 값만 돌려받는 배열 메소드를 이용하여 프로그래밍하시오. 모든 값을 돌려받도록 한다.
    nums.filter((num) => num % 2 == 1);
  
    // 4. 다음 arr 배열 값 중 3000과 같은 과일명을 보이시오.
    const arr = [
      { name: "apple", price: 1000 },
      { name: "orange", price: 3000 },
      { name: "banana", price: 400 },
    ];
    document.write(arr.find((fruit) => fruit.price == 3000).name + "<br>");
  
    // 5. 다음 배열 nums 값을 배열메소드를 이용하여 역순으로 출력하시오.
    const nums_ = [1, 6, 7, 4, 5, 11, 9, 8];
    document.write(nums_.reverse() + "<br>");
  
    // 6. nums 의 값을 다 더하기 위해 reduce메소드를 이용하여 프로그래밍하시오.
    document.write(nums.reduce((acc, cur) => acc + cur) + "<br>");
