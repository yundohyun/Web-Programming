    <script>
      function q(num) {
        const v = document.querySelector(`#q${num}`);
        v.textContent = ["백설공주", "둘다 문제다"][num - 1];
        v.style.color = ["orange", "red"][num - 1];
        v.style.fontSize = ["36px", "56px"][num - 1];
      }
    </script>
