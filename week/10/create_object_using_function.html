Create Object Using Function

      
    function createMonster(name, hp, att) {
      return { name, hp, att };
    }
    const m1 = createMonster("슬라임", 25, 10);
    const m2 = createMonster("고블린", 50, 20);
    const m3 = createMonster("드래곤", 100, 40);
    document.write(JSON.stringify(m1) + "<br>");
    document.write(JSON.stringify(m2) + "<br>");
    document.write(JSON.stringify(m3) + "<br>");
