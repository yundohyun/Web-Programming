    class Company {
      constructor(name, distance, pay) {
        this.name = name;
        this.distance = distance;
        this.pay = pay;
      }
    }
    const tesla = new Company("Tesla", 100, 1000);
    const samsung = new Company("Samsung", 200, 2000);
    document.write(
      `name : ${tesla.name}, distance : ${tesla.distance}, pay : ${tesla.pay}<br>`
    );
    document.write(
      `name : ${samsung.name}, distance : ${samsung.distance}, pay : ${samsung.pay}<br>`
    );
  
    document.write("<hr>");
