    // 객체 정렬
    const animal = [
      { name: "cat", weight: 5 },
      { name: "dog", weight: 10 },
      { name: "monkey", weight: 20 },
      { name: "lion", weight: 200 },
      { name: "elephant", weight: 1000 },
    ];
  
    const result2 = animal.sort(() => 1);
    document.write("result2 : ");
    result2.forEach((item) => document.write(`${item.name} `));
    document.write("<br>");
  
    const result3 = animal.sort(() => -1);
    document.write("result3 : ");
    result3.forEach((item) => document.write(`${item.name} `));
    document.write("<br>");
