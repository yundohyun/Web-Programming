    const num = [3, 8, 10, 2, 6, 5, 1, 4, 7, 9];
    const result = num.reduce((sum, i) => sum + i, 0);

    const cart = [
      { name: "orange", price: 5000 },
      { name: "apple", price: 10000 },
      { name: "banana", price: 15000 },
      { name: "melon", price: 20000 },
    ];
    const price = cart.reduce((sum, i) => sum + i.price, 0);
